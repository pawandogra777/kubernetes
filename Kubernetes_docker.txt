kind: Deployment
metadata:
  name: eob-pipeline
spec:
  replicas: 1
  selector:
    matchLabels:
      app: eob
  template:
    metadata:
      labels:
        app: eob
    spec:
      containers:
      - name: eob-container
        image: eob-pipeline
        imagePullPolicy: Never
